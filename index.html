Follow these 4 steps to connect your free Google Sheet database. Your data will sync across all devices forever — no subscription needed.

Extensions → Apps Script → Delete existing code → Paste this:

const SHEET = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; function doPost(e){try{const d=JSON.parse(e.postData.contents);if(d.action==='save'){const r=[d.id,d.name,d.age,d.sex,d.opd,d.mobile,d.hosp,d.diag,d.d1,d.d2,d.said,d.rec,d.pend,d.notes,d.ts];const rows=SHEET.getDataRange().getValues();const idx=rows.findIndex(r=>r[0]==d.id);if(idx>0)SHEET.getRange(idx+1,1,1,r.length).setValues([r]);else SHEET.appendRow(r);return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);}if(d.action==='delete'){const rows=SHEET.getDataRange().getValues();const idx=rows.findIndex(r=>r[0]==d.id);if(idx>0)SHEET.deleteRow(idx+1);return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);}}catch(err){return ContentService.createTextOutput(JSON.stringify({ok:false,err:err.message})).setMimeType(ContentService.MimeType.JSON);}} function doGet(e){const rows=SHEET.getDataRange().getValues();const keys=['id','name','age','sex','opd','mobile','hosp','diag','d1','d2','said','rec','pend','notes','ts'];const data=rows.slice(1).map(r=>Object.fromEntries(keys.map((k,i)=>[k,r[i]])));return ContentService.createTextOutput(JSON.stringify({ok:true,data})).setMimeType(ContentService.MimeType.JSON);}
